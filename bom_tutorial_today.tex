\begin{frame}{The BOM Template: Section by Section}
    \textbf{Your \texttt{mvp\_project\_spec.tex} has 3 critical sections:}
    
    \vspace{0.3cm}
    \begin{enumerate}
        \item \textbf{Requirements from PES} (copy 3 most hardware-relevant)
        \begin{itemize}
            \item These drive your component choices
            \item Example: ``R1: Measure soil moisture every 10 min with $\pm$2\% accuracy''
        \end{itemize}
        
        \vspace{0.2cm}
        \item \textbf{Bill of Materials} (the shopping list)
        \begin{itemize}
            \item Component, Part \#, Supplier, Unit \$, Qty, Have/Need
            \item \textbf{Critical column:} ``Why this part? (link to R1--R3)''
        \end{itemize}
        
        \vspace{0.2cm}
        \item \textbf{Datasheet citations} (proof of viability)
        \begin{itemize}
            \item URL or document name + section
            \item Example: ``DHT22 datasheet section 3.2: accuracy $\pm$0.5°C''
        \end{itemize}
    \end{enumerate}
    
    \vspace{0.2cm}
    \begin{block}{Grading Emphasis}
        \textbf{Justification} (why this part?) weighs more than just listing components.\\
        A 3-component BOM with excellent justification $>$ 10-component BOM with no reasoning.
    \end{block}
\end{frame}

\begin{frame}[shrink=12]{Common Mistakes to Avoid}
    \begin{columns}
        \column{0.5\textwidth}
        \textbf{Mistake 1: No justification}
        \begin{itemize}
            \item Bad: ``DHT22 temperature sensor''
            \item Fix: ``DHT22: $\pm$0.5°C (DS p.3) meets R1. DS18B20 needs OneWire (complex). BME280 $\pm$1°C insufficient.''
        \end{itemize}
        
        \vspace{0.15cm}
        \textbf{Mistake 2: No datasheet citation}
        \begin{itemize}
            \item Bad: ``ESP32 has enough memory''
            \item Fix: ``ESP32 DS Table 2: 520kB SRAM. Code: 180kB. Margin: 65\%.''
        \end{itemize}
        
        \column{0.5\textwidth}
        \textbf{Mistake 3: Ignoring scalability}
        \begin{itemize}
            \item Bad: BOM \$48 (barely under budget)
            \item Fix: ``Prototype: \$48. 1000 units: \$32 (bulk).''
        \end{itemize}
        
        \vspace{0.15cm}
        \textbf{Mistake 4: Single-source}
        \begin{itemize}
            \item Bad: ``Custom sensor from Alibaba''
            \item Fix: ``Primary: Sensor X (Digi-Key). Backup: Sensor Y (Mouser).''
        \end{itemize}
    \end{columns}

        \small
